```javascript
    markerIndex++;
    // Create marker with markerIndex
```

```javascript

    [startLat, startLng],              // Start location
    ...spots.map(s => [s.lat, s.lng]), // Intermediate spots
    [endLat, endLng]                   // End location
];

// Create Leaflet polyline
L.polyline(routePoints, {
    color: '#1e3c72',    // Navy blue
```
